    Uses a doubly linked list for O(1) access to most recent items and a hashmap for O(1) lookups.
    Expired items are lazily evicted on access.

        """Add node to the head of the list (most recently used)."""
        node.prev = None
        node.next = self.head

        if self.head is not None:
            self.head.prev = node

        self.head = node

        if self.tail is None:
            self.tail = node

        """Remove node from the linked list."""
        if node.prev is not None:
            node.prev.next = node.next
        else:
            self.head = node.next

            node.next.prev = node.prev
        else:
            self.tail = node.prev

        self._remove_node(node)
        self._add_to_head(node)

        """Remove and return the tail node (least recently used)."""
        if self.tail is None:
            return None

        node = self.tail
        self._remove_node(node)
        return node
